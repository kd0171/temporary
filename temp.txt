# text_normalization.py
import re
import unicodedata
from functools import lru_cache
from typing import List, Tuple, Optional

import numpy as np
import pandas as pd

import config


# -----------------------------
# 略語辞書の読み込みと適用
# -----------------------------

def load_abbrev_patterns(csv_path: Optional[str] = None) -> List[Tuple[re.Pattern, str]]:
    """
    略語辞書 CSV を読み込み、(正規表現, 置換文字列) のリストとして返す。
    CSV フォーマット例:
        pattern,replacement
        HPT,high pressure turbine
        LPT,low pressure turbine
    """
    if csv_path is None:
        csv_path = config.ABBREV_CSV

    csv_path = str(csv_path)
    df = pd.read_csv(csv_path)

    if "pattern" not in df.columns or "replacement" not in df.columns:
        raise ValueError("abbrev_patterns.csv には 'pattern', 'replacement' 列が必要です。")

    patterns: List[Tuple[re.Pattern, str]] = []
    for _, row in df.iterrows():
        # 単語境界でマッチさせる（大文字小文字は normalize 側で吸収）
        pat_str = r"\b" + re.escape(str(row["pattern"]).strip().lower()) + r"\b"
        compiled = re.compile(pat_str)
        replacement = str(row["replacement"]).strip().lower()
        patterns.append((compiled, replacement))
    return patterns


@lru_cache(maxsize=1)
def _cached_abbrev_patterns() -> List[Tuple[re.Pattern, str]]:
    """デフォルトパスの略語辞書をキャッシュ付きで取得"""
    try:
        return load_abbrev_patterns()
    except FileNotFoundError:
        # 略語辞書が無くても動くように空リスト
        return []


def apply_abbrev_to_text(text: str,
                         patterns: Optional[List[Tuple[re.Pattern, str]]] = None) -> str:
    """
    テキストに略語辞書を適用する。
    patterns が None の場合はデフォルトの CSV を使用。
    """
    if not isinstance(text, str):
        text = "" if pd.isna(text) else str(text)

    if patterns is None:
        patterns = _cached_abbrev_patterns()

    lowered = text.lower()
    for pat, repl in patterns:
        lowered = pat.sub(repl, lowered)
    return lowered


# -----------------------------
# s 問題を含む簡易レマタイザ
# -----------------------------

def _simple_lemma(token: str) -> str:
    """
    非常に簡易なレマタイザ。
    - 長さ > 3 で末尾が "ies" → "y" (e.g. "bodies" -> "body")
    - 長さ > 3 で末尾が "es" → "e.g. "cracks" -> "crack" など
    - 長さ > 3 で末尾が "s" → "s" を削除
    """
    if len(token) <= 3:
        return token

    if token.endswith("ies") and len(token) > 4:
        # e.g. "parties" -> "party"
        return token[:-3] + "y"

    if token.endswith("es") and len(token) > 4:
        # e.g. "cracks" ではなく "blades" 誤処理がありうるが簡易版として許容
        return token[:-2]

    if token.endswith("s"):
        return token[:-1]

    return token


def _remove_punctuation(text: str) -> str:
    """
    記号を削除し、英数字・アンダーバー・スペースのみ残す。
    """
    # Unicode 正規化（全角 → 半角など）
    text = unicodedata.normalize("NFKC", text)

    # アルファベット/数字/スペース/アンダーバー 以外をスペースに
    text = re.sub(r"[^\w\s]", " ", text)
    # アンダーバーも一旦スペース扱いにする（最後に _ に再結合する）
    text = text.replace("_", " ")
    # 複数スペースを1つに
    text = re.sub(r"\s+", " ", text).strip()
    return text


def normalize_text_basic(text: str,
                         use_abbrev: bool = True,
                         patterns: Optional[List[Tuple[re.Pattern, str]]] = None) -> str:
    """
    基本的な正規化:
      - NaN/None を空文字に
      - 小文字化
      - 略語辞書で展開 (任意)
      - 記号除去
      - 簡易レマタイザ（s 問題対応）
    """
    if not isinstance(text, str):
        text = "" if pd.isna(text) else str(text)

    # 小文字化 + Unicode 正規化
    text = unicodedata.normalize("NFKC", text).lower()

    # 略語展開（"hpt" -> "high pressure turbine" など）
    if use_abbrev:
        text = apply_abbrev_to_text(text, patterns=patterns)

    # 記号除去
    text = _remove_punctuation(text)

    if not text:
        return ""

    tokens = text.split()
    lemmas = [_simple_lemma(tok) for tok in tokens]
    return " ".join(lemmas)


def to_reason_key(text: str,
                  use_abbrev: bool = True,
                  patterns: Optional[List[Tuple[re.Pattern, str]]] = None) -> str:
    """
    normalize_text_basic() の結果をさらに
    スペース -> アンダーバー に変換して 1 トークン化したキーを返す。
    """
    normalized = normalize_text_basic(text, use_abbrev=use_abbrev, patterns=patterns)
    if not normalized:
        return ""
    return normalized.replace(" ", "_")


def add_reason_key(df: pd.DataFrame,
                   text_col: str = None,
                   reason_key_col: str = None,
                   abbrev_csv_path: Optional[str] = None) -> pd.DataFrame:
    """
    DataFrame に reason_key 列を追加する。
    - text_col: 元のテキスト列名（例: "removal_reason"）
    - reason_key_col: 追加する列名（例: "reason_key"）
    """
    if text_col is None:
        text_col = config.TEXT_COLUMN
    if reason_key_col is None:
        reason_key_col = config.REASON_KEY_COLUMN

    df = df.copy()

    # 略語辞書パターンの指定があれば（なければデフォルトを使用）
    patterns = None
    if abbrev_csv_path is not None:
        patterns = load_abbrev_patterns(abbrev_csv_path)

    # ベクトル化しやすいように numpy array 経由で処理
    texts = df[text_col].astype("object").to_numpy()

    keys = [
        to_reason_key(text, use_abbrev=True, patterns=patterns)
        for text in texts
    ]

    df[reason_key_col] = keys
    return df

# create_embeddings.py
"""
reason_key ごとの埋め込みを生成して保存するスクリプト。

実行例:
    python create_embeddings.py
"""

import json
from typing import List

import numpy as np
import pandas as pd
from sentence_transformers import SentenceTransformer

import config
from text_normalization import add_reason_key


def create_embeddings(
    input_csv: str | None = None,
    text_col: str | None = None,
    reason_key_col: str | None = None,
) -> None:
    if input_csv is None:
        input_csv = str(config.INPUT_CSV)
    if text_col is None:
        text_col = config.TEXT_COLUMN
    if reason_key_col is None:
        reason_key_col = config.REASON_KEY_COLUMN

    print(f"[create_embeddings] Loading data from: {input_csv}")
    df = pd.read_csv(input_csv)

    # reason_key を付与
    print("[create_embeddings] Adding reason_key column...")
    df = add_reason_key(df, text_col=text_col, reason_key_col=reason_key_col)

    # ユニークな reason_key を抽出
    unique_keys: List[str] = (
        df[reason_key_col]
        .fillna("")
        .astype(str)
        .replace("", np.nan)
        .dropna()
        .unique()
        .tolist()
    )

    print(f"[create_embeddings] Unique reason_key count: {len(unique_keys)}")

    # SentenceTransformer モデルのロード
    print(f"[create_embeddings] Loading SentenceTransformer: {config.SENTENCE_MODEL_NAME}")
    model = SentenceTransformer(config.SENTENCE_MODEL_NAME)

    # 埋め込み生成
    print("[create_embeddings] Encoding reason keys...")
    embeddings = model.encode(
        unique_keys,
        batch_size=64,
        show_progress_bar=True,
        convert_to_numpy=True,
        normalize_embeddings=True,  # コサイン類似度を使う場合に便利
    )

    # 保存
    print(f"[create_embeddings] Saving embeddings to {config.EMBEDDINGS_NPY}")
    np.save(config.EMBEDDINGS_NPY, embeddings)

    print(f"[create_embeddings] Saving reason_keys to {config.REASON_KEYS_NPY}")
    np.save(config.REASON_KEYS_NPY, np.array(unique_keys, dtype=object))

    # 参考: reason_key を含むデータを保存しておく
    output_csv = str(config.DATA_DIR / "removal_reasons_with_key.csv")
    print(f"[create_embeddings] Saving dataframe with reason_key to {output_csv}")
    df.to_csv(output_csv, index=False)

    print("[create_embeddings] Done.")


if __name__ == "__main__":
    create_embeddings()

# cluster_reviews.py
"""
埋め込み (embeddings.npy) に対して K-means を実行し、
- kmeans.joblib (モデル)
- cluster_meta.json (cluster_id -> cluster_name)
- reason_dict.json (reason_key -> {cluster_id, cluster_name})
などを保存するスクリプト。

実行例:
    python cluster_reviews.py --n_clusters 20
"""

import argparse
import json

import joblib
import numpy as np
from sklearn.cluster import KMeans
from sklearn.decomposition import PCA
import matplotlib.pyplot as plt

import config


def train_kmeans(n_clusters: int = 20) -> None:
    print("[cluster_reviews] Loading embeddings...")
    embeddings = np.load(config.EMBEDDINGS_NPY)
    reason_keys = np.load(config.REASON_KEYS_NPY, allow_pickle=True)

    print(f"[cluster_reviews] embeddings shape: {embeddings.shape}")
    print(f"[cluster_reviews] number of reason_keys: {len(reason_keys)}")

    # KMeans 学習
    print(f"[cluster_reviews] Training KMeans with n_clusters={n_clusters}...")
    kmeans = KMeans(
        n_clusters=n_clusters,
        random_state=config.RANDOM_SEED,
        n_init="auto",
        max_iter=300,
    )
    kmeans.fit(embeddings)

    # モデル保存
    print(f"[cluster_reviews] Saving KMeans model to {config.KMEANS_MODEL}")
    joblib.dump(kmeans, config.KMEANS_MODEL)

    # 各 reason_key のクラスタ割当
    labels = kmeans.predict(embeddings)

    # cluster_id -> cluster_name (初期値は "cluster_00" など)
    cluster_meta = {
        int(cid): f"cluster_{int(cid):02d}"
        for cid in range(n_clusters)
    }

    print(f"[cluster_reviews] Saving cluster_meta to {config.CLUSTER_META_JSON}")
    with open(config.CLUSTER_META_JSON, "w", encoding="utf-8") as f:
        json.dump(cluster_meta, f, ensure_ascii=False, indent=2)

    # reason_key -> {cluster_id, cluster_name}
    reason_dict = {}
    for key, cid in zip(reason_keys, labels):
        cid = int(cid)
        reason_dict[str(key)] = {
            "cluster_id": cid,
            "cluster_name": cluster_meta.get(cid, f"cluster_{cid:02d}"),
        }

    print(f"[cluster_reviews] Saving reason_dict to {config.REASON_DICT_JSON}")
    with open(config.REASON_DICT_JSON, "w", encoding="utf-8") as f:
        json.dump(reason_dict, f, ensure_ascii=False, indent=2)

    # 任意: PCA 2 次元プロットを保存
    try:
        print("[cluster_reviews] Creating PCA visualization (optional)...")
        pca = PCA(n_components=2, random_state=config.RANDOM_SEED)
        emb_2d = pca.fit_transform(embeddings)

        plt.figure(figsize=(8, 6))
        plt.scatter(emb_2d[:, 0], emb_2d[:, 1], c=labels, s=10, alpha=0.7)
        plt.title("KMeans clusters (PCA 2D)")
        plt.xlabel("PC1")
        plt.ylabel("PC2")
        plt.tight_layout()

        plot_path = config.MODELS_DIR / "clusters_pca.png"
        plt.savefig(plot_path, dpi=150)
        plt.close()
        print(f"[cluster_reviews] Saved PCA plot to {plot_path}")
    except Exception as e:
        print(f"[cluster_reviews] PCA visualization failed (ignored): {e}")

    print("[cluster_reviews] Done.")


if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--n_clusters", type=int, default=20)
    args = parser.parse_args()

    train_kmeans(n_clusters=args.n_clusters)

"""
推論用モジュール。

- predict_reason_category(text) -> dict
- add_reason_category_to_df(df, text_col=...) -> df

Dash からもインポートして利用できます。

import json
from functools import lru_cache
from typing import Dict, Optional, Tuple

import joblib
import numpy as np
import pandas as pd
from sentence_transformers import SentenceTransformer

import config
from text_normalization import to_reason_key


# -----------------------------
# ロード系（キャッシュ付き）
# -----------------------------

@lru_cache(maxsize=1)
def _load_sentence_model() -> SentenceTransformer:
    print(f"[inference] Loading SentenceTransformer: {config.SENTENCE_MODEL_NAME}")
    model = SentenceTransformer(config.SENTENCE_MODEL_NAME)
    return model


@lru_cache(maxsize=1)
def _load_kmeans():
    print(f"[inference] Loading KMeans model from {config.KMEANS_MODEL}")
    return joblib.load(config.KMEANS_MODEL)


@lru_cache(maxsize=1)
def _load_cluster_meta() -> Dict[int, str]:
    try:
        with open(config.CLUSTER_META_JSON, "r", encoding="utf-8") as f:
            meta = json.load(f)
        return {int(k): str(v) for k, v in meta.items()}
    except FileNotFoundError:
        return {}


@lru_cache(maxsize=1)
def _load_reason_dict() -> Dict[str, Dict[str, object]]:
    try:
        with open(config.REASON_DICT_JSON, "r", encoding="utf-8") as f:
            return json.load(f)
    except FileNotFoundError:
        return {}


# -----------------------------
# コア推論ロジック
# -----------------------------

def _predict_from_embedding_vector(
    vec: np.ndarray,
) -> Tuple[int, str]:
    """
    埋め込みベクトルから cluster_id, cluster_name を得る。
    """
    kmeans = _load_kmeans()
    cluster_meta = _load_cluster_meta()

    vec = vec.reshape(1, -1)
    cid = int(kmeans.predict(vec)[0])
    cname = cluster_meta.get(cid, f"cluster_{cid:02d}")
    return cid, cname


def _predict_from_reason_key(
    reason_key: str,
) -> Tuple[Optional[int], Optional[str], str]:
    """
    reason_key からクラスタを推論する。
    戻り値: (cluster_id, cluster_name, source)
        source: "dict" or "model" or "none"
    """
    reason_key = str(reason_key or "").strip()
    if not reason_key:
        return None, None, "none"

    reason_dict = _load_reason_dict()

    # 1) 既存辞書にあればそれを優先
    if reason_key in reason_dict:
        info = reason_dict[reason_key]
        cid = int(info["cluster_id"])
        cname = str(info["cluster_name"])
        return cid, cname, "dict"

    # 2) 無ければ Transformer + KMeans で推論
    #    埋め込みは normalized reason_key 文字列に対して行う
    model = _load_sentence_model()
    emb = model.encode(
        [reason_key],
        convert_to_numpy=True,
        normalize_embeddings=True,
    )[0]

    cid, cname = _predict_from_embedding_vector(emb)
    return cid, cname, "model"


def predict_reason_category(text: str) -> Dict[str, object]:
    """
    1 テキストに対して reason_key, cluster_id, cluster_name を返す関数。
    戻り値例:
        {
            "reason_key": "high_pressure_turbine_blade_crack",
            "cluster_id": 0,
            "cluster_name": "blade_damage",
            "source": "dict"  # or "model" or "none"
        }
    """
    # まず正規化して reason_key を作る
    reason_key = to_reason_key(text, use_abbrev=True)

    cid, cname, source = _predict_from_reason_key(reason_key)

    return {
        "reason_key": reason_key,
        "cluster_id": cid,
        "cluster_name": cname,
        "source": source,
    }

def add_reason_category_to_df(
    df: pd.DataFrame,
    text_col: str | None = None,
    reason_key_col: str | None = None,
    cluster_id_col: str | None = None,
    cluster_name_col: str | None = None,
    reason_category_col: str | None = None,
) -> pd.DataFrame:
    """
    DataFrame に reason_key / cluster_id / cluster_name / reason_category 列を追加する。

    - reason_category は cluster_name と同じ内容を持たせる想定。
    - 内部で重複する reason_key はまとめて推論し、パフォーマンスを確保する。
    """
    if text_col is None:
        text_col = config.TEXT_COLUMN
    if reason_key_col is None:
        reason_key_col = config.REASON_KEY_COLUMN
    if cluster_id_col is None:
        cluster_id_col = config.CLUSTER_ID_COLUMN
    if cluster_name_col is None:
        cluster_name_col = config.CLUSTER_NAME_COLUMN
    if reason_category_col is None:
        reason_category_col = config.REASON_CATEGORY_COLUMN

    df = df.copy()

    # 1) reason_key を付与（既にあれば再利用）
    if reason_key_col not in df.columns:
        from text_normalization import add_reason_key as _add_reason_key
        df = _add_reason_key(df, text_col=text_col, reason_key_col=reason_key_col)

    # 2) ユニークな reason_key ごとにクラスタを推論
    reason_keys = (
        df[reason_key_col]
        .fillna("")
        .astype(str)
        .tolist()
    )
    unique_keys = sorted(set(reason_keys))

    key_to_result: Dict[str, Dict[str, object]] = {}
    for key in unique_keys:
        if key not in key_to_result:
            cid, cname, source = _predict_from_reason_key(key)
            key_to_result[key] = {
                "cluster_id": cid,
                "cluster_name": cname,
                "source": source,
            }

    # 3) DataFrame に展開
    cluster_ids = []
    cluster_names = []
    for key in reason_keys:
        info = key_to_result.get(key, {"cluster_id": None, "cluster_name": None})
        cluster_ids.append(info["cluster_id"])
        cluster_names.append(info["cluster_name"])

    df[cluster_id_col] = cluster_ids
    df[cluster_name_col] = cluster_names
    df[reason_category_col] = cluster_names  # Dash で使うカテゴリー列

    return df

# text_normalization.py
import re
import unicodedata
from functools import lru_cache
from typing import List, Tuple, Optional

import numpy as np
import pandas as pd

import config


# -----------------------------
# 略語辞書
# -----------------------------

def load_abbrev_patterns(csv_path: Optional[str] = None) -> List[Tuple[re.Pattern, str]]:
    """
    abbrev_patterns.csv を読み込み、(regex, replacement) のリストとして返す。

    CSV 例:
        pattern,replacement
        HPT,high pressure turbine
        LPT,low pressure turbine
    """
    if csv_path is None:
        csv_path = config.ABBREV_CSV

    df = pd.read_csv(csv_path)

    if "pattern" not in df.columns or "replacement" not in df.columns:
        raise ValueError("abbrev_patterns.csv には 'pattern', 'replacement' 列が必要です。")

    patterns: List[Tuple[re.Pattern, str]] = []
    for _, row in df.iterrows():
        pat = str(row["pattern"]).strip().lower()
        repl = str(row["replacement"]).strip().lower()
        # 単語境界でマッチ
        pat_re = re.compile(r"\b" + re.escape(pat) + r"\b")
        patterns.append((pat_re, repl))
    return patterns


@lru_cache(maxsize=1)
def _cached_abbrev_patterns() -> List[Tuple[re.Pattern, str]]:
    try:
        return load_abbrev_patterns()
    except FileNotFoundError:
        return []


def apply_abbrev_to_text(text: str,
                         patterns: Optional[List[Tuple[re.Pattern, str]]] = None) -> str:
    if not isinstance(text, str):
        text = "" if pd.isna(text) else str(text)

    lowered = unicodedata.normalize("NFKC", text).lower()
    if patterns is None:
        patterns = _cached_abbrev_patterns()

    for pat, repl in patterns:
        lowered = pat.sub(repl, lowered)
    return lowered


# -----------------------------
# s 問題を含む簡易レマタイザ
# -----------------------------

def _simple_lemma(token: str) -> str:
    if len(token) <= 3:
        return token

    if token.endswith("ies") and len(token) > 4:
        return token[:-3] + "y"

    if token.endswith("es") and len(token) > 4:
        return token[:-2]

    if token.endswith("s"):
        return token[:-1]

    return token


def _remove_punctuation(text: str) -> str:
    text = unicodedata.normalize("NFKC", text)
    text = re.sub(r"[^\w\s]", " ", text)
    text = text.replace("_", " ")
    text = re.sub(r"\s+", " ", text).strip()
    return text


def normalize_text_basic(text: str,
                         use_abbrev: bool = True,
                         patterns: Optional[List[Tuple[re.Pattern, str]]] = None) -> str:
    """
    - 小文字化
    - 略語展開
    - 記号除去
    - s 問題の簡易レマタイザ
    """
    if not isinstance(text, str):
        text = "" if pd.isna(text) else str(text)

    text = unicodedata.normalize("NFKC", text).lower()

    if use_abbrev:
        text = apply_abbrev_to_text(text, patterns=patterns)

    text = _remove_punctuation(text)
    if not text:
        return ""

    tokens = text.split()
    lemmas = [_simple_lemma(tok) for tok in tokens]
    return " ".join(lemmas)


def to_reason_key(text: str,
                  use_abbrev: bool = True,
                  patterns: Optional[List[Tuple[re.Pattern, str]]] = None) -> str:
    normalized = normalize_text_basic(text, use_abbrev=use_abbrev, patterns=patterns)
    if not normalized:
        return ""
    return normalized.replace(" ", "_")


def add_reason_key(df: pd.DataFrame,
                   text_col: str | None = None,
                   reason_key_col: str | None = None,
                   abbrev_csv_path: Optional[str] = None) -> pd.DataFrame:
    """
    DataFrame に reason_key 列を追加する。
    """
    if text_col is None:
        text_col = config.TEXT_COLUMN
    if reason_key_col is None:
        reason_key_col = config.REASON_KEY_COLUMN

    df = df.copy()

    patterns = None
    if abbrev_csv_path is not None:
        patterns = load_abbrev_patterns(abbrev_csv_path)

    texts = df[text_col].astype("object").to_numpy()

    keys = [
        to_reason_key(text, use_abbrev=True, patterns=patterns)
        for text in texts
    ]

    df[reason_key_col] = keys
    return df

# cluster_tfidf.py
"""
TF-IDF + KMeans でクラスタを作り、
クラスタの中身を探索するための情報を CSV に落とすスクリプト。

実行例:
    python cluster_tfidf.py --n_clusters 20
"""

import argparse
import json

import joblib
import numpy as np
import pandas as pd
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.cluster import KMeans

import config
from text_normalization import add_reason_key


def run_clustering_tfidf(n_clusters: int = 20,
                         min_df: int = 3,
                         max_features: int = 5000) -> None:
    print(f"[cluster_tfidf] Loading data from {config.INPUT_CSV}")
    df = pd.read_csv(config.INPUT_CSV)

    # 1) reason_key 付与（正規化）
    print("[cluster_tfidf] Adding reason_key...")
    df = add_reason_key(df, text_col=config.TEXT_COLUMN,
                        reason_key_col=config.REASON_KEY_COLUMN)

    # 2) TF-IDF ベクトル化
    print("[cluster_tfidf] Vectorizing with TF-IDF...")
    vectorizer = TfidfVectorizer(
        analyzer="word",
        ngram_range=(1, 2),
        min_df=min_df,
        max_features=max_features,
    )
    X = vectorizer.fit_transform(df[config.REASON_KEY_COLUMN].fillna(""))

    print(f"[cluster_tfidf] TF-IDF matrix shape: {X.shape}")

    # 3) KMeans クラスタリング
    print(f"[cluster_tfidf] Fitting KMeans (n_clusters={n_clusters})...")
    kmeans = KMeans(
        n_clusters=n_clusters,
        random_state=config.RANDOM_SEED,
        n_init="auto",
        max_iter=300,
    )
    kmeans.fit(X)
    labels = kmeans.labels_

    df[config.CLUSTER_ID_COLUMN] = labels

    # 4) モデル保存（後で Dash で使いたくなったときのため）
    print(f"[cluster_tfidf] Saving TF-IDF vectorizer to {config.TFIDF_VECTORIZER_PATH}")
    joblib.dump(vectorizer, config.TFIDF_VECTORIZER_PATH)

    print(f"[cluster_tfidf] Saving KMeans model to {config.KMEANS_MODEL_PATH}")
    joblib.dump(kmeans, config.KMEANS_MODEL_PATH)

    # 5) クラスタ探索用 CSV を作る
    print("[cluster_tfidf] Preparing cluster exploration summary...")

    terms = np.array(vectorizer.get_feature_names_out())
    centers = kmeans.cluster_centers_
    order_centroids = centers.argsort(axis=1)[:, ::-1]  # 各クラスタで重みが大きい順

    rows = []
    for cid in range(n_clusters):
        cluster_mask = (labels == cid)
        cluster_df = df[cluster_mask]

        # 上位単語
        top_term_indices = order_centroids[cid][:15]
        top_terms = terms[top_term_indices].tolist()
        top_terms_str = ", ".join(top_terms)

        # 代表サンプル（元テキスト）をいくつか
        sample_texts = cluster_df[config.TEXT_COLUMN].dropna().astype(str).head(10).tolist()
        sample_texts_str = " || ".join(sample_texts)

        rows.append({
            "cluster_id": cid,
            "n_samples": int(cluster_mask.sum()),
            "top_terms": top_terms_str,
            "sample_texts": sample_texts_str,
        })

    explore_df = pd.DataFrame(rows)
    print(f"[cluster_tfidf] Saving exploration CSV to {config.CLUSTER_EXPLORE_CSV}")
    explore_df.to_csv(config.CLUSTER_EXPLORE_CSV, index=False)

    # 6) 初期の cluster_meta.json（名前は仮）
    cluster_meta = {int(cid): f"cluster_{cid:02d}" for cid in range(n_clusters)}
    print(f"[cluster_tfidf] Saving initial cluster_meta to {config.CLUSTER_META_JSON}")
    with open(config.CLUSTER_META_JSON, "w", encoding="utf-8") as f:
        json.dump(cluster_meta, f, ensure_ascii=False, indent=2)

    # 7) クラスタ付き DataFrame も保存（任意）
    output_csv = config.DATA_DIR / "removal_reasons_with_cluster_id.csv"
    print(f"[cluster_tfidf] Saving dataframe with cluster_id to {output_csv}")
    df.to_csv(output_csv, index=False)

    print("[cluster_tfidf] Done.")


if __name__ == "__main__":

    parser.add_argument("--n_clusters", type=int, default=20)
    parser.add_argument("--min_df", type=int, default=3)

# preprocess_for_dash.py
"""
クラスタ ID と cluster_meta.json に基づき、
cluster_name / reason_category 列を付けた CSV を作るスクリプト。

実行例:
    python preprocess_for_dash.py \
        --input data/removal_reasons_with_cluster_id.csv \
        --output data/removal_reasons_with_category.csv
"""
import argparse
import json

import pandas as pd

import config


def add_cluster_name_and_category(
    df: pd.DataFrame,
    cluster_meta_path: str | None = None,
) -> pd.DataFrame:
    if cluster_meta_path is None:
        cluster_meta_path = config.CLUSTER_META_JSON

    with open(cluster_meta_path, "r", encoding="utf-8") as f:
        cluster_meta = json.load(f)

    # 数値・文字列両方に対応
    cluster_meta_int = {int(k): str(v) for k, v in cluster_meta.items()}

    df = df.copy()
    df[config.CLUSTER_NAME_COLUMN] = df[config.CLUSTER_ID_COLUMN].map(cluster_meta_int)
    df[config.REASON_CATEGORY_COLUMN] = df[config.CLUSTER_NAME_COLUMN]
    return df


def main(input_csv: str, output_csv: str) -> None:
    print(f"[preprocess_for_dash] Loading: {input_csv}")
    df = pd.read_csv(input_csv)

    print("[preprocess_for_dash] Adding cluster_name / reason_category...")
    df = add_cluster_name_and_category(df)

    print(f"[preprocess_for_dash] Saving to: {output_csv}")
    df.to_csv(output_csv, index=False)

    print("[preprocess_for_dash] Done.")


if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument(
        "--input",
        type=str,
        default=str(config.DATA_DIR / "removal_reasons_with_cluster_id.csv"),
    )
    parser.add_argument(
        "--output",
        type=str,
        default=str(config.DATA_DIR / "removal_reasons_with_category.csv"),
    )
    args = parser.parse_args()

    main(args.input, args.output)
5. Dash から使うための関数（別ファイル）
「データを取得するコンポーネントで test_df としてデータを取得しており、その中の文字列を含む列にカテゴリー列を上記の前処理、カテゴリー化をした上で追加するような関数を別ファイルに」

バッチ処理前提なら「Dash ではすでに reason_category 付き CSV を読む」だけで済みますが、
「Dash 内でも同じロジックで一発で付けたい」こともあると思うので、ラッパー関数も用意しておきます。

※ここでは「新規データ推論」というより「そのときの test_df に対して TF-IDF + KMeans を Fit & Predict する一括処理」です（モデル保存・再利用はしない版に変更してもOK）。

ここでは、すでに TF-IDF & KMeans を学習済み（cluster_tfidf.py 済み）とし、
保存済みモデル & cluster_meta.json を使って test_df にカテゴリを付ける関数にしておきます。
（将来的に「真の推論」が欲しくなったら、そのまま再利用できます）

--- code ---

# reason_category_pipeline.py
"""
Dash から利用するための薄いラッパー。

test_df に対し、既存の TF-IDF + KMeans + cluster_meta を使って
cluster_id / cluster_name / reason_category を付ける。
"""

from typing import Optional
import json

import joblib
import pandas as pd

import config
from text_normalization import add_reason_key


def enrich_with_reason_category(
    df: pd.DataFrame,
    text_col: Optional[str] = None,
) -> pd.DataFrame:
    """
    取得した test_df に、
    reason_key / cluster_id / cluster_name / reason_category を追加する。

    ※ 前提:
        - models/tfidf_vectorizer.joblib
        - models/kmeans_tfidf.joblib
        - models/cluster_meta.json
      が存在し、クラスター名が設定済であること。
    """
    if text_col is None:
        text_col = config.TEXT_COLUMN

    df = df.copy()

    # 1) reason_key 付与
    df = add_reason_key(df, text_col=text_col,
                        reason_key_col=config.REASON_KEY_COLUMN)

    # 2) モデル & メタ読み込み
    vectorizer = joblib.load(config.TFIDF_VECTORIZER_PATH)
    kmeans = joblib.load(config.KMEANS_MODEL_PATH)

    with open(config.CLUSTER_META_JSON, "r", encoding="utf-8") as f:
        cluster_meta = json.load(f)
    cluster_meta_int = {int(k): str(v) for k, v in cluster_meta.items()}

    # 3) TF-IDF 化 & 予測
    X = vectorizer.transform(df[config.REASON_KEY_COLUMN].fillna(""))
    labels = kmeans.predict(X)
    df[config.CLUSTER_ID_COLUMN] = labels

    # 4) クラスタ名 & カテゴリ列
    df[config.CLUSTER_NAME_COLUMN] = df[config.CLUSTER_ID_COLUMN].map(cluster_meta_int)
    df[config.REASON_CATEGORY_COLUMN] = df[config.CLUSTER_NAME_COLUMN]

    return df

6. まとめ：KMeans 後の「探索フェーズ」は必須？
はい、（ほぼ）必須だと思っておく方が安全です。

KMeans は「この点群を n 個の塊に分けました」しか教えてくれない

「クラスタ 0 はブレード損傷、クラスタ 1 はオイル系統」といった意味付けは、
各クラスタに含まれるサンプルを人間が眺めて決めるプロセス が必要

そのために

上位 TF-IDF 単語 (top_terms)

代表サンプル (sample_texts)
を吐き出す cluster_explore.csv のようなものを作っておくのが定番パターン

今回の構成だと：

cluster_tfidf.py でクラスタリング & cluster_explore.csv 作成

cluster_explore.csv を見ながら cluster_meta.json を編集してカテゴリー名を決める

preprocess_for_dash.py で reason_category 付き CSV を作成

Dash でその CSV を読み込み、reason_category をフィルタに使う

という運用になります。

もし「cluster_explore.csv の中身をもう少しリッチにしたい（クラスタごとの件数ランキングや可視化など）」という要望が出てきたら、その部分だけ拡張してあげれば OK です。
